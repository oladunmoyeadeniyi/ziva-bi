## CONTEXT

This brief covers five related changes:

1. **Source model redesign** — replace the single `value_source` string with a
   `dimension_sources` JSONB array that stores multiple sources per dimension,
   each with an optional filter. Manual values are always available — not a source.

2. **Add form v2** — source-first flow replaced with a 3-step form:
   Step 1: Name (dropdown of standard names + "Not on list" custom),
   Step 2: Connected sources (multi-select checkboxes with optional org filter),
   Step 3: Review & save.

3. **Reactivate toggle** — currently the toggle only deactivates. It must also
   reactivate a dimension when toggled back on.

4. **Hard delete for custom dimensions** — standard dimensions can only be
   deactivated (soft delete). Custom dimensions (not in the standard codes list)
   can be permanently deleted.

5. **Seed fix** — update the seed endpoint to use the correct old codes
   (`real_internal_order`, `statistical_internal_order`) and fix existing
   dimensions with wrong `value_source`.

```python
@router.patch("/dimensions/{dimension_id}/reactivate",
              response_model=DimensionResponse)
async def reactivate_dimension(
    dimension_id: uuid.UUID,
    current_user: CurrentUser = Depends(require_auth),
    db: AsyncSession = Depends(get_db),
) -> DimensionResponse:
    """Reactivate a soft-deleted dimension."""
    tenant_id = _require_tenant(current_user)
    _require_admin(current_user)

    result = await db.execute(
        select(TenantDimension).where(
            TenantDimension.id == dimension_id,
            TenantDimension.tenant_id == tenant_id,
        )
    )
    dim = result.scalar_one_or_none()
    if not dim:
        raise HTTPException(
            status_code=status.HTTP_404_NOT_FOUND,
            detail="Dimension not found."
        )
    dim.is_active = True
    await db.flush()
    await db.refresh(dim)
    return DimensionResponse.from_orm(dim)
```

### 4c. Add hard delete endpoint for custom dimensions

Add after the reactivate endpoint:

```python
STANDARD_DIMENSION_CODES = {
    "cost_center", "material", "statistical_order", "statistical_internal_order",
    "real_order", "real_internal_order", "customer_order", "employee",
    "brand", "region", "channel", "project",
}

@router.delete("/dimensions/{dimension_id}/permanent",
               status_code=status.HTTP_204_NO_CONTENT)
async def hard_delete_dimension(
    dimension_id: uuid.UUID,
    current_user: CurrentUser = Depends(require_auth),
    db: AsyncSession = Depends(get_db),
) -> None:
    """
    Permanently delete a custom dimension and all its values.
    Standard dimensions cannot be permanently deleted — use soft delete instead.
    """
    tenant_id = _require_tenant(current_user)
    _require_admin(current_user)

    result = await db.execute(
        select(TenantDimension).where(
            TenantDimension.id == dimension_id,
            TenantDimension.tenant_id == tenant_id,
        )
    )
    dim = result.scalar_one_or_none()
    if not dim:
        raise HTTPException(
            status_code=status.HTTP_404_NOT_FOUND,
            detail="Dimension not found."
        )

    if dim.code in STANDARD_DIMENSION_CODES:
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail="Standard dimensions cannot be permanently deleted. Use deactivate instead.",
        )

    # Delete all values first
    await db.execute(
        delete(DimensionValue).where(
            DimensionValue.dimension_id == dimension_id,
            DimensionValue.tenant_id == tenant_id,
        )
    )

    # Delete the dimension
    await db.delete(dim)
    await db.flush()
```

```tsx
      </div>
    )}
  </div>
)}
```
