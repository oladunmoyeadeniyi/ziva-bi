### 4b. Add endpoint to fetch saved dimension requirements for a GL account

Add this endpoint BEFORE the existing `/coa/{gl_id}/dimensions` PATCH:

```python
@router.get("/coa/{gl_id}/dimensions")
async def get_coa_dimensions(
    gl_id: uuid.UUID,
    current_user: CurrentUser = Depends(require_auth),
    db: AsyncSession = Depends(get_db),
) -> list[dict]:
    """Return saved dimension requirements for a GL account."""
    tenant_id = _require_tenant(current_user)

    result = await db.execute(
        select(GLDimensionRequirement).where(
            GLDimensionRequirement.gl_account_id == gl_id,
            GLDimensionRequirement.tenant_id == tenant_id,
        )
    )
    reqs = result.scalars().all()
    return [
        {
            "dimension_id": str(r.dimension_id),
            "requirement": r.requirement,
        }
        for r in reqs
    ]
```
