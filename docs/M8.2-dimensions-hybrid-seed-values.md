## CONTEXT

Four fixes to the Dimensions page:

1. **Hybrid source redesign** — hybrid should allow the user to pick WHICH
   auto-sources to combine (not hardcoded to employee + manual). Options:
   employee master, org structure. Manual codes are always included in hybrid.

2. **Seed standard dimensions** — the 5 standard dimensions (Cost center,
   Material/SKU, Statistical IO, Real IO, Customer order) should be seeded
   for the tenant if they don't exist, with correct `value_source` values.
   This is a backend endpoint called once on first visit to the Dimensions page.

3. **Fix existing dimension sources** — existing dimensions created before
   `value_source` column existed show `manual` incorrectly. Update:
   - Code `cost_center` → `value_source = "org_structure"`
   - Code `statistical_order` or `statistical_internal_order` → `value_source = "hybrid"`
   - Code `material` → `value_source = "product_master"`
   - Code `customer_order` → `value_source = "customer_order"`
   - Code `real_order` or `real_statistical_order` or `real_internal_order` → keep `manual`

4. **Wire values buttons** — Add value, Download template, and Bulk upload
   buttons must be functional for manual dimensions.

        "description": "Tracks actual costs by project or initiative.",
        "icon": "git-commit",
        "sort_order": 4,
    },
    {
        "name": "Customer order",
        "code": "customer_order",
        "is_required": False,
        "value_source": "customer_order",
        "description": "Customer categories/segments. Manual now, auto-syncs from customer master when AR is active.",
        "icon": "users-group",
        "sort_order": 5,
    },
]


@router.post("/dimensions/seed-standard", status_code=200)
async def seed_standard_dimensions(
    current_user: CurrentUser = Depends(require_auth),
    db: AsyncSession = Depends(get_db),
) -> dict:
    """
    Seed the standard dimensions for a tenant if they don't exist yet.
    Also fixes value_source for existing dimensions that were created before
    the value_source column was added.
    Called by the frontend on first visit to the Dimensions page.
    """
    _require_admin(current_user)
    tenant_id = _require_tenant(current_user)

    seeded = 0
    fixed = 0

    # Fix existing dimensions that have wrong value_source
    SOURCE_FIXES = {
        "cost_center": "org_structure",
        "material": "product_master",
        "statistical_order": "hybrid",
        "statistical_internal_order": "hybrid",
        "real_order": "manual",
        "real_statistical_order": "manual",
        "real_internal_order": "manual",
        "customer_order": "customer_order",
    }

    existing_result = await db.execute(
        select(TenantDimension).where(TenantDimension.tenant_id == tenant_id)
    )
    existing_dims = existing_result.scalars().all()
    existing_codes = {d.code for d in existing_dims}

    for dim in existing_dims:
        correct_source = SOURCE_FIXES.get(dim.code)
        if correct_source and (not dim.value_source or dim.value_source == "manual" and correct_source != "manual"):
            dim.value_source = correct_source
            fixed += 1

    # Seed missing standard dimensions
    for std in STANDARD_DIMENSIONS:
        if std["code"] not in existing_codes:
            new_dim = TenantDimension(
                tenant_id=tenant_id,
                name=std["name"],
                code=std["code"],
                is_required=std["is_required"],
                value_source=std["value_source"],
                description=std["description"],
                icon=std["icon"],
                sort_order=std["sort_order"],
                is_active=True,
            )
            db.add(new_dim)
            seeded += 1

    await db.commit()
    return {"seeded": seeded, "fixed": fixed}
```

@router.get("/dimensions/{dimension_id}/values/template")
async def download_dimension_values_template(
    dimension_id: uuid.UUID,
    current_user: CurrentUser = Depends(require_auth),
    db: AsyncSession = Depends(get_db),
):
    """Download a blank XLSX template for bulk uploading dimension values."""
    _require_admin(current_user)
    await _get_dimension_or_404(dimension_id, _require_tenant(current_user), db)

    try:
        import openpyxl
        from openpyxl.styles import Font, PatternFill, Alignment
    except ImportError:
        raise HTTPException(status_code=500, detail="openpyxl not installed.")

    wb = openpyxl.Workbook()
    ws = wb.active
    ws.title = "Values"

    # Header row
    headers = ["Code *", "Name *", "Description"]
    for col, h in enumerate(headers, 1):
        cell = ws.cell(row=1, column=col, value=h)
        cell.font = Font(name="Arial", bold=True, size=11, color="FFFFFF")
        cell.fill = PatternFill("solid", fgColor="1E3A5F")
        cell.alignment = Alignment(horizontal="center")

    ws.column_dimensions["A"].width = 20
    ws.column_dimensions["B"].width = 30
    ws.column_dimensions["C"].width = 40

    # Sample row
    ws.cell(row=2, column=1, value="CC-001").font = Font(name="Arial", size=10, italic=True, color="888888")
    ws.cell(row=2, column=2, value="Example value name").font = Font(name="Arial", size=10, italic=True, color="888888")
    ws.cell(row=2, column=3, value="Optional description").font = Font(name="Arial", size=10, italic=True, color="888888")

    ws.freeze_panes = "A2"

    buf = io.BytesIO()
    wb.save(buf)
    buf.seek(0)

    return StreamingResponse(
        buf,
        media_type="application/vnd.openxmlformats-officedocument.spreadsheetml.sheet",
        headers={"Content-Disposition": f"attachment; filename=dimension_values_template.xlsx"},
    )

const loadDimValues = async (dimId: string) => {
  if (!accessToken) return;
  try {
    const vals = await apiFetch<DimensionValue[]>(
      `/api/config/dimensions/${dimId}/values`,
      { token: accessToken }
    );
    setDimValues(prev => ({ ...prev, [dimId]: vals }));
  } catch {}
};

const handleAddValue = async (dimId: string) => {
  if (!accessToken || !addValueCode.trim() || !addValueName.trim()) return;
  setAddingValue(true);
  setAddValueError(null);
  try {
    await apiFetch(`/api/config/dimensions/${dimId}/values`, {
      method: "POST",
      token: accessToken,
      body: JSON.stringify({
        code: addValueCode.trim(),
        name: addValueName.trim(),
        description: addValueDesc.trim() || undefined,
      }),
    });
    setAddValueDimId(null);
    setAddValueCode("");
    setAddValueName("");
    setAddValueDesc("");
    await loadDimValues(dimId);
  } catch (err) {
    setAddValueError(err instanceof Error ? err.message : "Failed to add value.");
  } finally {
    setAddingValue(false);
  }
};

const handleDownloadTemplate = (dimId: string) => {
  const url = `${process.env.NEXT_PUBLIC_API_URL || "http://localhost:8000"}/api/config/dimensions/${dimId}/values/template`;
  window.open(url, "_blank");
};

const handleBulkUpload = async (dimId: string, file: File) => {
  if (!accessToken) return;
  const formData = new FormData();
  formData.append("file", file);
  try {
    await fetch(
      `${process.env.NEXT_PUBLIC_API_URL || "http://localhost:8000"}/api/config/dimensions/${dimId}/values/upload`,
      {
        method: "POST",
        headers: { Authorization: `Bearer ${accessToken}` },
        body: formData,
      }
    );
    await loadDimValues(dimId);
  } catch (err) {
    setError("Bulk upload failed.");
  }
};
```

## NOTES FOR CC

- The `seed-standard` endpoint is idempotent — safe to call on every page load.
  It only creates dimensions that don't exist and only fixes sources that are wrong.
- The hybrid sources are stored in the `description` field as `hybrid:employee_master`
  or `hybrid:employee_master,org_structure`. This is a pragmatic approach until
  a dedicated `hybrid_sources` JSONB column is added in a future migration.
- The `handleDownloadTemplate` function opens the template URL directly — this
  requires the backend to NOT require auth headers for this endpoint, OR use
  a fetch-based download. Use the fetch approach with auth header for consistency.
  Replace `window.open` with a fetch that creates a blob download link.
- The bulk upload uses the existing `POST /api/config/dimensions/{id}/values/upload`
  endpoint which already exists in config.py.
- `StreamingResponse` and `io` imports are already in config.py — no new imports
  needed for the template download endpoint.
