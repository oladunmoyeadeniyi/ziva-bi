   Source: manual now, auto-sync from group structure when Intercompany is built.

3. **Seed fix** — use upsert logic (check by code + tenant_id) so the seed
   never creates duplicates. Also fix the STANDARD_CODES set.

4. **Deactivated section collapsed by default** — show a toggle button instead
   of always rendering deactivated dims.

5. **Values inline on Master data / values tab** — for org_structure dimensions,
   fetch and show the actual cost center list right there. For employee_master,
   show employee list. Always include a "Go to source" secondary link.

6. **Customer order duplicate fix** — the DB currently has 2 active customer_order
   records. The seed must deduplicate on load.

### 4c. Add endpoint to fetch org structure values for a dimension

Add this endpoint BEFORE the existing `/dimensions/{dimension_id}/values` endpoint:

```python
@router.get("/dimensions/{dimension_id}/inline-values")
async def get_dimension_inline_values(
    dimension_id: uuid.UUID,
    current_user: CurrentUser = Depends(require_auth),
    db: AsyncSession = Depends(get_db),
) -> dict:
    """
    Return inline values for a dimension based on its source type.
    For org_structure: returns cost center nodes.
    For employee_master: returns employee list.
    For manual/product_master/customer_master: returns stored dimension values.
    """
    _require_admin(current_user)
    tenant_id = _require_tenant(current_user)

    dim = await _get_dimension_or_404(dimension_id, tenant_id, db)

    sources = dim.dimension_sources or []
    source_types = [s.get("source_type") for s in sources]

    result_data = {
        "dimension_id": str(dimension_id),
        "value_source": dim.value_source,
        "sections": [],
    }

    # Org structure values
    if "org_structure" in source_types:
        src_filter = next(
            (s.get("filter") for s in sources if s.get("source_type") == "org_structure"),
            None
        )
        query = select(OrgStructureNode).where(
            OrgStructureNode.tenant_id == tenant_id,
            OrgStructureNode.node_type == "Cost center",
            OrgStructureNode.is_active.is_(True),
        )
        if src_filter and src_filter.get("parent_code"):
            parent_result = await db.execute(
                select(OrgStructureNode).where(
                    OrgStructureNode.tenant_id == tenant_id,
                    OrgStructureNode.code == src_filter["parent_code"],
                )
            )
            parent = parent_result.scalar_one_or_none()
            if parent:
                query = query.where(OrgStructureNode.parent_id == parent.id)
        nodes_result = await db.execute(query.order_by(OrgStructureNode.sort_order))
        nodes = nodes_result.scalars().all()
        result_data["sections"].append({
            "source_type": "org_structure",
            "label": "Cost centers (from org structure)",
            "read_only": True,
            "items": [
                {
                    "code": n.cost_center_code or n.code,
                    "name": n.name,
                    "parent_id": str(n.parent_id) if n.parent_id else None,
                }
                for n in nodes
            ],
        })

    # Employee master values
    if "employee_master" in source_types:
        emp_result = await db.execute(
            select(Employee).where(
                Employee.tenant_id == tenant_id,
                Employee.is_active.is_(True),
            ).order_by(Employee.full_name)
        )
        employees = emp_result.scalars().all()
        result_data["sections"].append({
            "source_type": "employee_master",
            "label": "Employee codes (from employee master)",
            "read_only": True,
            "items": [
                {
                    "code": e.employee_code or str(e.id)[:8],
                    "name": e.full_name,
                }
                for e in employees
            ],
        })

    # Manual / stored values (always include)
    manual_result = await db.execute(
        select(DimensionValue).where(
            DimensionValue.dimension_id == dimension_id,
            DimensionValue.tenant_id == tenant_id,
            DimensionValue.is_active.is_(True),
        ).order_by(DimensionValue.sort_order)
    )
    manual_values = manual_result.scalars().all()
    if manual_values:
        result_data["sections"].append({
            "source_type": "manual",
            "label": "Manual values",
            "read_only": False,
            "items": [
                {
                    "id": str(v.id),
                    "code": v.code,
                    "name": v.name,
                    "description": v.description,
                }
                for v in manual_values
            ],
        })

    return result_data
```

Add loader function:
```typescript
const loadInlineValues = async (dimId: string) => {
  if (!accessToken || inlineValues[dimId] || loadingValues.has(dimId)) return;
  setLoadingValues(prev => new Set(prev).add(dimId));
  try {
    const data = await apiFetch<InlineValues>(
      `/api/config/dimensions/${dimId}/inline-values`,
      { token: accessToken }
    );
    setInlineValues(prev => ({ ...prev, [dimId]: data }));
  } catch {}
  finally {
    setLoadingValues(prev => { const next = new Set(prev); next.delete(dimId); return next; });
  }
};
```

Call `loadInlineValues` when a dimension card is expanded:
```typescript
const toggleExpand = (id: string, dim?: Dimension) => {
  setExpandedIds(prev => {
    const next = new Set(prev);
    if (next.has(id)) {
      next.delete(id);
    } else {
      next.add(id);
      if (dim) loadInlineValues(id);
    }
    return next;
  });
};
```
